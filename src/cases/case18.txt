class BoolCheck is
	method OperationCheck1(a : Boolean, x : Boolean) : Boolean is
	var b : Boolean
		if a.ToInteger.Equal(1) then
			b := a.And(x)
		else
			b := a.Or(x)
		end
		return b
	end

	method OperationCheck2(a : Boolean, x : Boolean) : Boolean is
	var b : Boolean
		if a.ToInteger.Equal(1) then
			b := a.Xor(x)
		else
			b := x.Not()
		end
		return b
	end
end
